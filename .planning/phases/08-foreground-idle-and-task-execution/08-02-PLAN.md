```rust
// lettre 0.11 with "builder" feature
use lettre::message::{Message, MultiPart, SinglePart, Attachment, Body};
use lettre::transport::smtp::authentication::{Credentials, Mechanism};
use lettre::{AsyncSmtpTransport, AsyncTransport, Tokio1Executor};

// TLS config
AsyncSmtpTransport::<Tokio1Executor>::builder_dangerous(host)
    .tls(Tls::Wrapper(tls_params))   // direct TLS (port 465)
    .tls(Tls::Required(tls_params))  // STARTTLS (port 587)
    .tls(Tls::None)                  // cleartext (port 25)
    .credentials(Credentials::new(user, pass))
    .authentication(vec![Mechanism::Plain])  // or Mechanism::Xoauth2
    .timeout(Some(Duration::from_secs(30)))
    .build();

// MIME nesting: mixed -> alternative -> related
MultiPart::mixed()
    .multipart(MultiPart::alternative()
        .singlepart(SinglePart::plain(text))
        .multipart(MultiPart::related()
            .singlepart(SinglePart::html(html))
            .singlepart(Attachment::new_inline(cid).body(bytes, mime))))
    .singlepart(Attachment::new(filename).body(bytes, mime))
```

     pub struct DraftData {
         pub from: Vec<ContactField>,      // [{name, email}]
         pub to: Vec<ContactField>,
         pub cc: Vec<ContactField>,
         pub bcc: Vec<ContactField>,
         pub reply_to: Vec<ContactField>,

     a. Build message headers: From, To (multiple), Cc, Bcc, Subject, In-Reply-To, References, Reply-To
